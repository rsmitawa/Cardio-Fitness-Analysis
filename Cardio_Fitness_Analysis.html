# Plotting a pie chart for marital status

# Frequency table of marital status
marital_table = cardio_df['MaritalStatus'].value_counts()

# Function to calculate percentage of population for labeling
def func(pct, allvals):
    absolute = int(round(pct/100.*np.sum(allvals)))
    return "{:.1f}%\n({:d} g)".format(pct, absolute)

# Plotting pie chart using matplotlib
plt.pie(marital_table, 
        labels = marital_table.index,
        autopct=lambda pct: func(pct, marital_table), 
        shadow=True)
plt.title("Pie chart for marital status")
plt.show()
Insight¶

The pie chart shows the population demographic of customers with their marital status.
